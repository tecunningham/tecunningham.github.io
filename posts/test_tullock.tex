Test with r1=2, r2=2, c1=1, c2=1, V=1:
\pgfmathsetmacro{\rone}{2}
\pgfmathsetmacro{\rtwo}{2}

\pgfmathsetmacro{\ctwo}{1}
\pgfmathsetmacro{\V}{1}
\pgfmathsetmacro{\Kone}{\rone*\V/\cone}
\pgfmathsetmacro{\Ktwo}{\rtwo*\V/\ctwo}

K1 = \Kone, K2 = \Ktwo

At t=1:
\pgfmathsetmacro{\t}{1}
\pgfmathsetmacro{\xtworesult}{pow(\Kone*pow(\t,(\rone-1)/(\rone))/pow(1+\t,2), \rone/\rtwo)}
\pgfmathsetmacro{\xoneresult}{\Kone * \t / pow(1+\t,2)}

x2(1) = \xtworesult

x1(1) = \xoneresult
