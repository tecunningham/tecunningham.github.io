% 5 isovalue curves per panel, labeled by player 1's share (30,40,50,60,70)
% Args:
%   #1 = r_1, #2 = r_2, #3 = alpha_1, #4 = alpha_2
\newcommand{\TullockIsoContent}[4]{%
  \nextgroupplot[
    axis lines=left,
    xlabel={$x_2$},
    ylabel={$x_1$},
    xmin=0, xmax=100,
    ymin=0, ymax=100,
    title={$\substack{r_1=#1,\ r_2=#2\\ \alpha_1=#3,\ \alpha_2=#4}$},
    declare function={
      rone    = #1;
      rtwo    = #2;
      alphaone = #3;
      alphatwo = #4;
    }
  ]
  % Five isovalue curves for p = 0.3, 0.4, 0.5, 0.6, 0.7
  % p = 0.3  (label 30)
  \addplot+[thick, no markers, color=red,
            domain=0:{min(95.0,
                          pow(95.0 /
                              pow((0.3*alphatwo)/(alphaone*(1-0.3)), 1.0/rone),
                              1.0/(rtwo/rone)))},
            samples=200]
    { pow((0.3*alphatwo)/(alphaone*(1-0.3)), 1.0/rone) * pow(x, rtwo/rone) }
    node[pos=1, anchor=west, xshift=2pt] {$30$};

  % p = 0.4  (label 40)
  \addplot+[thick, no markers, color=orange,
            domain=0:{min(95.0,
                          pow(95.0 /
                              pow((0.4*alphatwo)/(alphaone*(1-0.4)), 1.0/rone),
                              1.0/(rtwo/rone)))},
            samples=200]
    { pow((0.4*alphatwo)/(alphaone*(1-0.4)), 1.0/rone) * pow(x, rtwo/rone) }
    node[pos=1, anchor=west, xshift=2pt] {$40$};

  % p = 0.5  (label 50)
  \addplot+[thick, no markers, color=yellow!80!black,
            domain=0:{min(95.0,
                          pow(95.0 /
                              pow((0.5*alphatwo)/(alphaone*(1-0.5)), 1.0/rone),
                              1.0/(rtwo/rone)))},
            samples=200]
    { pow((0.5*alphatwo)/(alphaone*(1-0.5)), 1.0/rone) * pow(x, rtwo/rone) }
    node[pos=1, anchor=west, xshift=2pt] {$50$};

  % p = 0.6  (label 60)
  \addplot+[thick, no markers, color=green!70!yellow,
            domain=0:{min(95.0,
                          pow(95.0 /
                              pow((0.6*alphatwo)/(alphaone*(1-0.6)), 1.0/rone),
                              1.0/(rtwo/rone)))},
            samples=200]
    { pow((0.6*alphatwo)/(alphaone*(1-0.6)), 1.0/rone) * pow(x, rtwo/rone) }
    node[pos=1, anchor=west, xshift=2pt] {$60$};

  % p = 0.7  (label 70)
  \addplot+[thick, no markers, color=green!80!black,
            domain=0:{min(95.0,
                          pow(95.0 /
                              pow((0.7*alphatwo)/(alphaone*(1-0.7)), 1.0/rone),
                              1.0/(rtwo/rone)))},
            samples=200]
    { pow((0.7*alphatwo)/(alphaone*(1-0.7)), 1.0/rone) * pow(x, rtwo/rone) }
    node[pos=1, anchor=west, xshift=2pt] {$70$};
}
